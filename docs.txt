# scale to 5 replicas in default namespace
kubectl scale deployment/finance-app --replicas=5

# scale in a specific namespace
kubectl scale deployment/finance-app --replicas=5 -n finance-namespace


#Reboot finance apps
kubectl rollout restart deployment/finance-app -n finance-namespace

#deploy new version of finance apps
kubectl set image deployment/finance-app finance-app-container=myrepo/finance-app:v2.0 -n finance-namespace
